#include <WiFi.h>
#include <PubSubClient.h>
#include <Wire.h>
#include <Adafruit_Sensor.h>

// --- ‡∏ï‡∏±‡πâ‡∏á‡∏Ñ‡πà‡∏≤ WiFi ---
const char* ssid = "YOUR_WIFI_SSID";       // ‡πÉ‡∏™‡πà‡∏ä‡∏∑‡πà‡∏≠ WiFi ‡∏Ç‡∏≠‡∏á‡∏Ñ‡∏∏‡∏ì
const char* password = "YOUR_WIFI_PASSWORD"; // ‡πÉ‡∏™‡πà‡∏£‡∏´‡∏±‡∏™‡∏ú‡πà‡∏≤‡∏ô WiFi

// --- ‡∏ï‡∏±‡πâ‡∏á‡∏Ñ‡πà‡∏≤ MQTT ---
const char* mqtt_server = "YOUR_NODE_RED_SERVER_IP"; // IP ‡∏Ç‡∏≠‡∏á Server ‡∏ó‡∏µ‡πà‡∏£‡∏±‡∏ô MQTT Broker
const int mqtt_port = 1883;                    // Port ‡∏°‡∏≤‡∏ï‡∏£‡∏ê‡∏≤‡∏ô‡∏Ç‡∏≠‡∏á MQTT
const char* mqtt_topic = "vecskill";           // Topic ‡∏ó‡∏µ‡πà‡∏ï‡πâ‡∏≠‡∏á‡∏Å‡∏≤‡∏£‡∏™‡πà‡∏á‡∏Ç‡πâ‡∏≠‡∏°‡∏π‡∏•‡πÑ‡∏õ

// --- ‡∏ï‡∏±‡πâ‡∏á‡∏Ñ‡πà‡∏≤ Sensor ---
Adafruit_BME280 bme; // ‡πÉ‡∏ä‡πâ‡∏á‡∏≤‡∏ô I2C
// ‡∏´‡∏°‡∏≤‡∏¢‡πÄ‡∏´‡∏ï‡∏∏: ‡∏ñ‡πâ‡∏≤‡πÉ‡∏ä‡πâ BMP280, ‡∏Ñ‡πà‡∏≤‡∏Ñ‡∏ß‡∏≤‡∏°‡∏ä‡∏∑‡πâ‡∏ô (Humidity) ‡∏à‡∏∞‡∏≠‡πà‡∏≤‡∏ô‡πÑ‡∏°‡πà‡πÑ‡∏î‡πâ

// --- ‡∏ï‡∏±‡∏ß‡πÅ‡∏õ‡∏£‡∏à‡∏±‡∏î‡∏Å‡∏≤‡∏£‡πÄ‡∏ß‡∏•‡∏≤ ---
unsigned long previousMillis = 0;
const long interval = 10000; // ‡∏™‡πà‡∏á‡∏Ç‡πâ‡∏≠‡∏°‡∏π‡∏•‡∏ó‡∏∏‡∏Å 10 ‡∏ß‡∏¥‡∏ô‡∏≤‡∏ó‡∏µ (10000 ms)

// --- Clients ---
WiFiClient espClient;
PubSubClient client(espClient);

/**
 * ‡∏ü‡∏±‡∏á‡∏Å‡πå‡∏ä‡∏±‡∏ô‡πÄ‡∏ä‡∏∑‡πà‡∏≠‡∏°‡∏ï‡πà‡∏≠ WiFi
 */
void setup_wifi() {
  delay(10);
  Serial.println();
  Serial.print("Connecting to ");
  Serial.println(ssid);
  WiFi.begin(ssid, password);
  while (WiFi.status() != WL_CONNECTED) {
    delay(500);
    Serial.print(".");
  }
  Serial.println("");
  Serial.println("WiFi connected");
  Serial.print("IP address: ");
  Serial.println(WiFi.localIP());
}

/**
 * ‡∏ü‡∏±‡∏á‡∏Å‡πå‡∏ä‡∏±‡∏ô‡πÄ‡∏ä‡∏∑‡πà‡∏≠‡∏°‡∏ï‡πà‡∏≠ MQTT Broker
 */
void reconnect_mqtt() {
  while (!client.connected()) {
    Serial.print("Attempting MQTT connection...");
    String clientId = "ESP32Client-";
    clientId += String(random(0xffff), HEX);

    if (client.connect(clientId.c_str())) {
      Serial.println("connected");
    } else {
      Serial.print("failed, rc=");
      Serial.print(client.state());
      Serial.println(" try again in 5 seconds");
      delay(5000); // ‡∏£‡∏≠ 5 ‡∏ß‡∏¥‡∏ô‡∏≤‡∏ó‡∏µ‡∏Å‡πà‡∏≠‡∏ô‡∏•‡∏≠‡∏á‡πÉ‡∏´‡∏°‡πà
    }
  }
}

/**
 * ‡∏ü‡∏±‡∏á‡∏Å‡πå‡∏ä‡∏±‡∏ô Setup ‡∏´‡∏•‡∏±‡∏Å
 */
void setup() {
  Serial.begin(115200);
  Wire.begin(); // ‡πÄ‡∏£‡∏¥‡πà‡∏°‡∏ï‡πâ‡∏ô I2C (SDA=21, SCL=22 ‡∏™‡∏≥‡∏´‡∏£‡∏±‡∏ö ESP32)

  setup_wifi();
  client.setServer(mqtt_server, mqtt_port);

  Serial.println("Initializing BME/BMP280 sensor...");
  if (!bme.begin(0x76)) { // ‡∏ï‡∏£‡∏ß‡∏à‡∏™‡∏≠‡∏ö Sensor ‡∏ó‡∏µ‡πà I2C address 0x76 (‡∏´‡∏£‡∏∑‡∏≠ 0x77)
    Serial.println("Could not find a valid BME/BMP280 sensor, check wiring!");
    while (1) delay(10);
  }
  Serial.println("Sensor found!");
}

/**
 * ‡∏ü‡∏±‡∏á‡∏Å‡πå‡∏ä‡∏±‡∏ô Loop ‡∏´‡∏•‡∏±‡∏Å
 */
void loop() {
  if (!client.connected()) {
    reconnect_mqtt();
  }
  client.loop(); // ‡πÉ‡∏´‡πâ MQTT client ‡∏ó‡∏≥‡∏á‡∏≤‡∏ô

  unsigned long currentMillis = millis();
  if (currentMillis - previousMillis >= interval) {
    previousMillis = currentMillis;

    // --- ‡∏≠‡πà‡∏≤‡∏ô‡∏Ñ‡πà‡∏≤‡∏à‡∏≤‡∏Å Sensor ---
    float temperature = bme.readTemperature();       // ‡∏≠‡∏á‡∏®‡∏≤‡πÄ‡∏ã‡∏•‡πÄ‡∏ã‡∏µ‡∏¢‡∏™
    float humidity = bme.readHumidity();         // %RH
    float pressure = bme.readPressure() / 100.0F; // hPa

    if (isnan(temperature) || isnan(pressure)) {
      Serial.println("Failed to read from BME/BMP sensor!");
      return;
    }
    if (isnan(humidity)) {
      humidity = 0.0; // BMP280 ‡πÑ‡∏°‡πà‡∏°‡∏µ‡πÄ‡∏ã‡πá‡∏ô‡πÄ‡∏ã‡∏≠‡∏£‡πå‡∏Ñ‡∏ß‡∏≤‡∏°‡∏ä‡∏∑‡πâ‡∏ô
    }

    // --- ‡∏™‡∏£‡πâ‡∏≤‡∏á‡∏Ç‡πâ‡∏≠‡∏Ñ‡∏ß‡∏≤‡∏° JSON ---
    char jsonPayload[256];
    snprintf(jsonPayload, sizeof(jsonPayload),
             "{\"temperature\":%.2f, \"humidity\":%.2f, \"pressure\":%.2f}",
             temperature, humidity, pressure);

    // --- ‡∏™‡πà‡∏á‡∏Ç‡πâ‡∏≠‡∏°‡∏π‡∏•‡πÑ‡∏õ‡∏¢‡∏±‡∏á MQTT Broker ---
    Serial.print("Publishing to topic: ");
    Serial.println(mqtt_topic);
    Serial.println(jsonPayload);

    if (client.publish(mqtt_topic, jsonPayload)) {
      Serial.println("Message published successfully");
    } else {
      Serial.println("Message publish failed");
    }
  }
}
