  WiFi.persistent(false);                         // ‡πÑ‡∏°‡πà‡πÄ‡∏Ç‡∏µ‡∏¢‡∏ô‡∏Ç‡πâ‡∏≠‡∏°‡∏π‡∏• Wi-Fi ‡∏•‡∏á‡πÅ‡∏ü‡∏•‡∏ä‡∏ó‡∏∏‡∏Å‡∏Ñ‡∏£‡∏±‡πâ‡∏á (‡∏¢‡∏∑‡∏î‡∏≠‡∏≤‡∏¢‡∏∏‡πÅ‡∏ü‡∏•‡∏ä)
  WiFi.begin(WIFI_SSID, WIFI_PASSWORD);           // ‡πÄ‡∏£‡∏¥‡πà‡∏°‡∏Å‡∏£‡∏∞‡∏ö‡∏ß‡∏ô‡∏Å‡∏≤‡∏£‡πÄ‡∏ä‡∏∑‡πà‡∏≠‡∏°‡∏ï‡πà‡∏≠ (‡πÑ‡∏°‡πà‡∏ö‡∏•‡πá‡∏≠‡∏Å loop)
  return false;                                   // ‡∏ì ‡∏à‡∏∏‡∏î‡∏ô‡∏µ‡πâ‡∏¢‡∏±‡∏á‡πÑ‡∏°‡πà‡πÄ‡∏ä‡∏∑‡πà‡∏≠‡∏° ‡∏à‡∏∂‡∏á‡∏Ñ‡∏∑‡∏ô false (‡∏•‡∏π‡∏õ‡∏à‡∏∞‡∏Å‡∏•‡∏±‡∏ö‡∏°‡∏≤‡πÄ‡∏ä‡πá‡∏Ñ‡∏≠‡∏µ‡∏Å)

// ---------------------- ‡∏ü‡∏±‡∏á‡∏Å‡πå‡∏ä‡∏±‡∏ô‡∏î‡∏π‡πÅ‡∏•‡∏Å‡∏≤‡∏£‡πÄ‡∏ä‡∏∑‡πà‡∏≠‡∏°‡∏ï‡πà‡∏≠ MQTT (non-blocking) ----------------------
bool ensureMQTT() {                               // ‡∏Ñ‡∏∑‡∏ô true ‡∏ñ‡πâ‡∏≤‡πÄ‡∏ä‡∏∑‡πà‡∏≠‡∏° MQTT ‡πÅ‡∏•‡πâ‡∏ß
  if (WiFi.status() != WL_CONNECTED) return false;// ‡∏ñ‡πâ‡∏≤ Wi-Fi ‡∏¢‡∏±‡∏á‡πÑ‡∏°‡πà‡∏û‡∏£‡πâ‡∏≠‡∏° ‡∏Å‡πá‡∏¢‡∏±‡∏á‡πÄ‡∏ä‡∏∑‡πà‡∏≠‡∏° MQTT ‡πÑ‡∏°‡πà‡πÑ‡∏î‡πâ
  if (mqtt.connected()) return true;              // ‡∏ñ‡πâ‡∏≤ MQTT ‡πÄ‡∏ä‡∏∑‡πà‡∏≠‡∏°‡πÅ‡∏•‡πâ‡∏ß ‡∏Å‡πá‡∏à‡∏ö (true)

  unsigned long now = millis();                   // ‡∏≠‡πà‡∏≤‡∏ô‡πÄ‡∏ß‡∏•‡∏≤ ‡∏ì ‡∏ï‡∏≠‡∏ô‡∏ô‡∏µ‡πâ
  if (now - lastMqttAttempt < MQTT_RETRY_MS)      // ‡∏ñ‡πâ‡∏≤‡∏¢‡∏±‡∏á‡πÑ‡∏°‡πà‡∏ñ‡∏∂‡∏á‡∏£‡∏≠‡∏ö‡∏ó‡∏µ‡πà‡∏ï‡πâ‡∏≠‡∏á‡∏•‡∏≠‡∏á‡πÄ‡∏ä‡∏∑‡πà‡∏≠‡∏°‡πÉ‡∏´‡∏°‡πà
    return false;                                 // ‡∏≠‡∏≠‡∏Å‡πÑ‡∏õ‡∏Å‡πà‡∏≠‡∏ô (false)

  lastMqttAttempt = now;                          // ‡∏ö‡∏±‡∏ô‡∏ó‡∏∂‡∏Å‡πÄ‡∏ß‡∏•‡∏≤‡∏û‡∏¢‡∏≤‡∏¢‡∏≤‡∏°‡∏•‡πà‡∏≤‡∏™‡∏∏‡∏î
  mqtt.setServer(MQTT_HOST, MQTT_PORT);           // ‡∏Å‡∏≥‡∏´‡∏ô‡∏î‡∏ó‡∏µ‡πà‡∏≠‡∏¢‡∏π‡πà‡πÅ‡∏•‡∏∞‡∏û‡∏≠‡∏£‡πå‡∏ï‡∏Ç‡∏≠‡∏á broker
  mqtt.setKeepAlive(30);                          // ‡∏ï‡∏±‡πâ‡∏á‡∏Ñ‡πà‡∏≤ keep-alive ‡πÄ‡∏õ‡πá‡∏ô 30 ‡∏ß‡∏¥‡∏ô‡∏≤‡∏ó‡∏µ (broker ‡∏à‡∏∞‡∏Ñ‡∏≤‡∏î‡∏´‡∏ß‡∏±‡∏á ping ‡∏†‡∏≤‡∏¢‡πÉ‡∏ô‡∏ä‡πà‡∏ß‡∏á‡∏ô‡∏µ‡πâ)
  mqtt.setSocketTimeout(30);                      // ‡∏ï‡∏±‡πâ‡∏á timeout ‡∏Å‡∏≤‡∏£‡∏≠‡πà‡∏≤‡∏ô/‡πÄ‡∏Ç‡∏µ‡∏¢‡∏ô socket 30 ‡∏ß‡∏¥‡∏ô‡∏≤‡∏ó‡∏µ

  char clientId[40];                              // ‡∏ö‡∏±‡∏ü‡πÄ‡∏ü‡∏≠‡∏£‡πå‡∏™‡∏≥‡∏´‡∏£‡∏±‡∏ö‡πÄ‡∏Å‡πá‡∏ö clientID (‡∏ï‡πâ‡∏≠‡∏á‡πÑ‡∏°‡πà‡∏ã‡πâ‡∏≥‡πÉ‡∏ô broker)
  uint64_t mac = ESP.getEfuseMac();               // ‡∏≠‡πà‡∏≤‡∏ô‡πÄ‡∏•‡∏Ç MAC ‡∏Ç‡∏≠‡∏á ESP32 (‡∏¢‡∏π‡∏ô‡∏µ‡∏Ñ)
  snprintf(clientId, sizeof(clientId),            // ‡∏™‡∏£‡πâ‡∏≤‡∏á clientID ‡∏à‡∏≤‡∏Å MAC (‡πÄ‡∏ä‡πà‡∏ô ESP32_ABCDEF12)
           "ESP32_%llX", mac);

  Serial.print(F("[MQTT] Connecting as "));       // ‡∏û‡∏¥‡∏°‡∏û‡πå‡∏ß‡πà‡∏≤‡∏Å‡∏≥‡∏•‡∏±‡∏á‡πÄ‡∏ä‡∏∑‡πà‡∏≠‡∏° MQTT
  Serial.print(clientId);                         // ‡∏û‡∏¥‡∏°‡∏û‡πå clientID
  Serial.print(F(" to "));                        // ‡∏Ç‡πâ‡∏≠‡∏Ñ‡∏ß‡∏≤‡∏°‡∏ï‡πà‡∏≠
  Serial.print(MQTT_HOST);                        // ‡∏û‡∏¥‡∏°‡∏û‡πå‡πÇ‡∏Æ‡∏™‡∏ï‡πå‡∏Ç‡∏≠‡∏á broker
  Serial.print(F(":"));                           // ‡∏Ñ‡∏±‡πà‡∏ô‡∏î‡πâ‡∏ß‡∏¢ colon
  Serial.println(MQTT_PORT);                      // ‡∏û‡∏¥‡∏°‡∏û‡πå‡∏û‡∏≠‡∏£‡πå‡∏ï‡∏Ç‡∏≠‡∏á broker

  // ‡πÄ‡∏ä‡∏∑‡πà‡∏≠‡∏° MQTT ‡∏û‡∏£‡πâ‡∏≠‡∏°‡∏ï‡∏±‡πâ‡∏á‡∏Ñ‡πà‡∏≤ Last Will & Testament (LWT)
  bool ok = mqtt.connect(                         // ‡πÄ‡∏£‡∏µ‡∏¢‡∏Å‡πÄ‡∏ä‡∏∑‡πà‡∏≠‡∏°‡∏ï‡πà‡∏≠‡πÑ‡∏õ‡∏¢‡∏±‡∏á broker
    clientId,                                     // clientID ‡∏ó‡∏µ‡πà‡∏¢‡∏π‡∏ô‡∏µ‡∏Ñ
    STATUS_TOPIC,                                 // willTopic: ‡∏´‡∏±‡∏ß‡∏Ç‡πâ‡∏≠‡∏ó‡∏µ‡πà‡∏à‡∏∞‡∏™‡πà‡∏á‡πÉ‡∏´‡πâ‡πÄ‡∏≠‡∏á‡∏ñ‡πâ‡∏≤‡∏Ñ‡∏•‡∏≤‡∏¢‡πÄ‡∏ä‡∏∑‡πà‡∏≠‡∏°‡∏ï‡πà‡∏≠‡πÅ‡∏ö‡∏ö‡∏ú‡∏¥‡∏î‡∏õ‡∏Å‡∏ï‡∏¥
    1,                                            // willQoS: ‡πÉ‡∏ä‡πâ QoS 1 ‡∏™‡∏≥‡∏´‡∏£‡∏±‡∏ö LWT
    true,                                         // willRetain: ‡πÄ‡∏Å‡πá‡∏ö retain ‡πÉ‡∏´‡πâ‡∏ú‡∏π‡πâ‡∏°‡∏≤‡∏ó‡∏µ‡∏´‡∏•‡∏±‡∏á‡πÄ‡∏´‡πá‡∏ô‡∏™‡∏ñ‡∏≤‡∏ô‡∏∞‡∏•‡πà‡∏≤‡∏™‡∏∏‡∏î
    "offline"                                     // willMessage: ‡∏Ç‡πâ‡∏≠‡∏Ñ‡∏ß‡∏≤‡∏° LWT ‡πÄ‡∏°‡∏∑‡πà‡∏≠‡∏´‡∏•‡∏∏‡∏î
  );

  if (ok) {                                       // ‡∏ñ‡πâ‡∏≤‡πÄ‡∏ä‡∏∑‡πà‡∏≠‡∏°‡∏ï‡πà‡∏≠‡∏™‡∏≥‡πÄ‡∏£‡πá‡∏à
    Serial.println(F("[MQTT] Connected"));        // ‡∏û‡∏¥‡∏°‡∏û‡πå‡πÉ‡∏ô Serial ‡∏ß‡πà‡∏≤‡∏™‡∏≥‡πÄ‡∏£‡πá‡∏à
    mqtt.publish(STATUS_TOPIC, "online", true);   // ‡∏õ‡∏£‡∏∞‡∏Å‡∏≤‡∏®‡∏ß‡πà‡∏≤ online (retain=true) ‡πÄ‡∏û‡∏∑‡πà‡∏≠‡∏≠‡∏±‡∏õ‡πÄ‡∏î‡∏ï‡∏™‡∏ñ‡∏≤‡∏ô‡∏∞
    // (‡∏ñ‡πâ‡∏≤‡∏ï‡πâ‡∏≠‡∏á‡∏Å‡∏≤‡∏£ subscribe ‡∏´‡∏±‡∏ß‡∏Ç‡πâ‡∏≠‡πÉ‡∏î ‡πÜ ‡πÉ‡∏´‡πâ‡πÄ‡∏û‡∏¥‡πà‡∏° mqtt.subscribe(...) ‡∏ó‡∏µ‡πà‡∏ô‡∏µ‡πà)
  } else {                                        // ‡∏ñ‡πâ‡∏≤‡πÄ‡∏ä‡∏∑‡πà‡∏≠‡∏°‡πÑ‡∏°‡πà‡∏™‡∏≥‡πÄ‡∏£‡πá‡∏à
    Serial.print(F("[MQTT] Failed, rc="));        // ‡∏û‡∏¥‡∏°‡∏û‡πå‡∏£‡∏´‡∏±‡∏™‡∏Ñ‡∏ß‡∏≤‡∏°‡∏ú‡∏¥‡∏î‡∏û‡∏•‡∏≤‡∏î
    Serial.println(mqtt.state());                 // mqtt.state() ‡πÅ‡∏™‡∏î‡∏á‡πÄ‡∏´‡∏ï‡∏∏‡∏ú‡∏• (‡πÄ‡∏ä‡πà‡∏ô -2, -4, ‡∏Ø‡∏•‡∏Ø)
  }
  return ok;                                      // ‡∏Ñ‡∏∑‡∏ô‡∏Ñ‡πà‡∏≤‡∏ú‡∏•‡∏•‡∏±‡∏û‡∏ò‡πå‡∏Å‡∏≤‡∏£‡πÄ‡∏ä‡∏∑‡πà‡∏≠‡∏°

// ---------------------- ‡∏ü‡∏±‡∏á‡∏Å‡πå‡∏ä‡∏±‡∏ô‡∏≠‡πà‡∏≤‡∏ô BME280 ‡πÅ‡∏•‡∏∞‡∏™‡πà‡∏á‡∏Ç‡∏∂‡πâ‡∏ô MQTT ----------------------
void publishTelemetry() {                         // ‡∏ü‡∏±‡∏á‡∏Å‡πå‡∏ä‡∏±‡∏ô‡∏™‡πà‡∏á‡∏Ç‡πâ‡∏≠‡∏°‡∏π‡∏•‡πÄ‡∏ó‡πÄ‡∏•‡πÄ‡∏°‡∏ó‡∏£‡∏µ (JSON)
  if (!bmeReady || !mqtt.connected()) return;     // ‡∏ñ‡πâ‡∏≤‡πÄ‡∏ã‡∏ô‡πÄ‡∏ã‡∏≠‡∏£‡πå‡πÑ‡∏°‡πà‡∏û‡∏£‡πâ‡∏≠‡∏° ‡∏´‡∏£‡∏∑‡∏≠ MQTT ‡∏¢‡∏±‡∏á‡πÑ‡∏°‡πà‡πÄ‡∏ä‡∏∑‡πà‡∏≠‡∏° ‡πÉ‡∏´‡πâ‡∏¢‡∏Å‡πÄ‡∏•‡∏¥‡∏Å

  float t = bme.readTemperature();                // ‡∏≠‡πà‡∏≤‡∏ô‡∏≠‡∏∏‡∏ì‡∏´‡∏†‡∏π‡∏°‡∏¥ (¬∞C)
  float h = bme.readHumidity();                   // ‡∏≠‡πà‡∏≤‡∏ô‡∏Ñ‡∏ß‡∏≤‡∏°‡∏ä‡∏∑‡πâ‡∏ô‡∏™‡∏±‡∏°‡∏û‡∏±‡∏ó‡∏ò‡πå (%RH)
  float p = bme.readPressure() / 100.0F;          // ‡∏≠‡πà‡∏≤‡∏ô‡∏Ñ‡∏ß‡∏≤‡∏°‡∏î‡∏±‡∏ô (Pa) ‡πÅ‡∏•‡πâ‡∏ß‡πÅ‡∏õ‡∏•‡∏á‡πÄ‡∏õ‡πá‡∏ô hPa ‡πÇ‡∏î‡∏¢‡∏´‡∏≤‡∏£ 100

  if (isnan(t) || isnan(h) || isnan(p)) {         // ‡∏õ‡πâ‡∏≠‡∏á‡∏Å‡∏±‡∏ô‡∏Å‡∏£‡∏ì‡∏µ‡∏≠‡πà‡∏≤‡∏ô‡∏Ñ‡πà‡∏≤‡πÑ‡∏î‡πâ NaN (‡∏≠‡πà‡∏≤‡∏ô‡∏û‡∏•‡∏≤‡∏î)
    Serial.println(F("[BME280] NAN reading, skip"));// ‡πÅ‡∏à‡πâ‡∏á‡πÄ‡∏ï‡∏∑‡∏≠‡∏ô‡πÅ‡∏•‡πâ‡∏ß‡πÑ‡∏°‡πà‡∏™‡πà‡∏á
    return;                                       // ‡∏≠‡∏≠‡∏Å‡∏à‡∏≤‡∏Å‡∏ü‡∏±‡∏á‡∏Å‡πå‡∏ä‡∏±‡∏ô
  }

  char payload[160];                              // ‡∏ö‡∏±‡∏ü‡πÄ‡∏ü‡∏≠‡∏£‡πå‡πÄ‡∏Å‡πá‡∏ö JSON ‡∏ó‡∏µ‡πà‡∏à‡∏∞‡∏™‡πà‡∏á
  snprintf(payload, sizeof(payload),              // ‡∏™‡∏£‡πâ‡∏≤‡∏á‡∏™‡∏ï‡∏£‡∏¥‡∏á JSON ‡∏î‡πâ‡∏ß‡∏¢ snprintf
           "{\"temperature\":%.2f,\"humidity\":%.2f,\"pressure\":%.2f,\"rssi\":%d}",
           t, h, p, WiFi.RSSI());                 // ‡∏ù‡∏±‡∏á‡∏Ñ‡πà‡∏≤‡∏ó‡∏µ‡πà‡∏ß‡∏±‡∏î‡πÑ‡∏î‡πâ‡πÅ‡∏•‡∏∞‡∏Ñ‡∏ß‡∏≤‡∏°‡πÅ‡∏£‡∏á‡∏™‡∏±‡∏ç‡∏ç‡∏≤‡∏ì Wi-Fi (RSSI)

  Serial.print(F("[PUB] "));                      // ‡∏û‡∏¥‡∏°‡∏û‡πå prefix ‡∏™‡∏≥‡∏´‡∏£‡∏±‡∏ö‡∏î‡∏π‡πÉ‡∏ô Serial
  Serial.println(payload);                        // ‡∏û‡∏¥‡∏°‡∏û‡πå‡πÄ‡∏ô‡∏∑‡πâ‡∏≠‡∏´‡∏≤ JSON ‡∏ó‡∏µ‡πà‡∏à‡∏∞‡∏™‡πà‡∏á

  mqtt.publish(TELE_TOPIC, payload, false);       // ‡∏™‡πà‡∏á‡∏Ç‡∏∂‡πâ‡∏ô MQTT ‡∏ó‡∏µ‡πà‡∏´‡∏±‡∏ß‡∏Ç‡πâ‡∏≠ telemetry (retain=false ‡∏™‡∏≥‡∏´‡∏£‡∏±‡∏ö‡∏Ç‡πâ‡∏≠‡∏°‡∏π‡∏•‡πÑ‡∏´‡∏•)

  digitalWrite(LED_PIN, HIGH); delay(20);         // ‡∏Å‡∏∞‡∏û‡∏£‡∏¥‡∏ö LED ‡∏™‡∏±‡πâ‡∏ô ‡πÜ ‡πÄ‡∏û‡∏∑‡πà‡∏≠‡∏¢‡∏∑‡∏ô‡∏¢‡∏±‡∏ô‡∏Å‡∏≤‡∏£‡∏™‡πà‡∏á
  digitalWrite(LED_PIN, LOW);                     // ‡∏õ‡∏¥‡∏î LED
}

  digitalWrite(LED_PIN, LOW);                     // ‡πÄ‡∏£‡∏¥‡πà‡∏°‡∏ï‡πâ‡∏ô‡πÉ‡∏´‡πâ LED ‡∏õ‡∏¥‡∏î‡πÑ‡∏ß‡πâ‡∏Å‡πà‡∏≠‡∏ô

    } else {                                      // ‡∏ñ‡πâ‡∏≤‡∏¢‡∏±‡∏á‡πÑ‡∏°‡πà‡∏û‡∏ö
